Off-the-Record Lib - twstrike/otr3


    type foo struct {
        bs []byte
    }

    func zeroes(n int) []byte {
        return make([]byte, n)
    }

    func wipeBytes(b []byte) {
        copy(b, zeroes(len(b)))
    }


    func main() {
        a := foo{bs: []byte{0x01, 0x02}}
        b := a.bs
        wipeBytes(b)
        fmt.Println(b)    //b == []byte{}
        fmt.Println(a.bs) //a.bs == []byte{}
    }
